open class LinkedList{
data class Node(
	var data: Int,
    var next: Node? = null
)
    var head: Node? = null
    var tail: Node? = null
    var size = 0
    
    fun addfirst(data: Int){
        val newNode = Node(data)
        size++
        println(newNode)
        if(head == null){
            head = newNode
            tail = newNode
            println("head = ${head}")
            return
        }
        newNode.next = head
        println("newNode next = ${newNode.next}")
        head = newNode
        println("newNode head = ${head}")
        
        println("Head Data = ${head?.data}")
        println("Head Next = ${head?.next}")
        
    }
    fun addLast(data: Int){
        val newNode = Node(data)
        size++
        println(newNode)
        if(head == null && tail == null){
            head = newNode
            tail = newNode
            println("head and tail IF= ${head} $tail")
            return
        }
        tail?.next = newNode
        println("tail next = ${tail?.next}")
        tail = newNode
        println("newNode = ${newNode}")
        
        println("Tail Data = ${tail?.data}")
        println("Tail Next = ${tail?.next}")
        
    }
    fun add(idx: Int, data: Int){
        if(idx == 0){
            addfirst(data)
            return
        }    
        val newNode = Node(data)
        size++
        println("NEW NODE = $newNode")
        var temp: Node? = head
        println("temp = $temp")
        var i = 0 
        println("i = $i")
       println("idx = $idx")
        
        while(i < idx-1){
            temp = temp?.next
            println("temp :next= $temp")
            i++
        }
        //i = idx-1 and temp ->prev
        newNode.next = temp?.next
        println("newNode.next ${newNode.next}")
        temp?.next = newNode
        println("temp?.next ${temp?.next}")
        
    }
    fun removefirst(){	// O(1)
       if (size == 0) {
       println("Linked List is Empty")
       }
        
    if (size == 1) {
        head = null
        tail = null
        size = 0
        return
    } 
        head = head?.next
    	size--
        println("head = ${head}")
        println("Head Data = ${head?.data}")
        println("Head Next = ${head?.next}")
        
    }
    fun removeLast(){ // O(n)
       if (size == 0) {
       println("Linked List is Empty")
       }
        
    if (size == 1) {
        head = null
        tail = null
        size = 0
        return
    } 
    
    //prev: i = size-2
    var prev = head
       for(i in 0 until size-2){
           prev = prev?.next
       }
       	val `value` = prev?.next?.data // tail.data
       	prev?.next = null
        tail = prev
    	size--
        println("head = ${head}")
        println("Head Data = ${head?.data}")
        println("Head Next = ${head?.next}")
        
    }
    
    fun search(key:Int):Int{ //O(n)
        var i = 0
        var temp = head
        while(temp !=null){
            if(temp.data == key){
                return i 
            }
            i++
            temp = temp.next
        }
        return -1
    
    }
    fun searchResursion(key : Int):Int{ //O(n)
        return helper(head,key)
    }
    fun helper(head: Node?, key : Int):Int{
        if(head == null){return -1}
        if(head?.data == key){
            return 0 
        }
        var idx = helper(head?.next, key)
        if(idx == -1){return -1}
        return idx+1
    }
    fun reverse(){
        var prev: Node? = null
        var tail = head
        var curr = tail
        var next : Node? = null
        while(curr !=null){
            next = curr?.next
            curr?.next = prev
            prev = curr
            curr = next
        }
        head = prev  
    }
    fun deleteNthFromEnd(n:Int){
        //calculate size
        var size = 0
        var temp = head
        // here we find size, althogue we alredy have size varible
        while(temp != null){
            temp = temp?.next
            size++
        }
        
        if(n == size){
            head = head?.next //remove first
            return
        }
        
        //size - n
       	var i = 1
        val iToFind = size-n
        var prev = head
        while(i < iToFind){
         prev = prev?.next
            i++
        }
        
        prev?.next = prev?.next?.next
        return
    }
    fun printll(){
        if(head == null){
            println("LinkedList is Empty")
            return
        }
        var temp: Node? = head
        println("HEad is : $temp")
        while(temp !=null){
            println(temp.data)
            temp = temp.next
        }
    }
}
