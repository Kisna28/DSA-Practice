import java.util.*

class Heap {
    private val heap = mutableListOf<Int>()

// Time Complexity: O(log n)
// Space Complexity: O(1) extra (excluding the heap storage itself)

fun add(data:Int){
    heap.add(data)
    
    var i = heap.size - 1
    
    while(i > 0){
        val parent = (i - 1) /2
        if(heap[i] < heap[parent]){
            val temp = heap[i]
            heap[i] = heap[parent]
            heap[parent] = temp
            i = parent
        }else{
            break
    }
    } 
}

// O(1)
fun getMin():Int{
    if(heap.isEmpty()){
     throw NoSuchElementException("Heap is empty") 
    }
    
    return heap[0]
}

fun printHeap() {
        println(heap)
    }
}

fun main() {
   val heap = Heap()

    heap.add(2)
    heap.add(3)
    heap.add(4)
    heap.add(5)
	heap.add(10)
	heap.add(1)
    
    heap.printHeap()
    
    println("Get min : ${heap.getMin()}")
}
