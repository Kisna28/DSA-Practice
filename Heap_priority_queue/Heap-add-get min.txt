fun heapSort(arr:IntArray){
    val n = arr.size
    for(i in n/2 -1 downTo 0){
        heapifyDownForSort(arr, n, i)
    }
    
   // Extract elements one by one

    for(i in n-1 downTo 0){
        val temp = arr[0]
        arr[0] = arr[i]
        arr[i] = temp
       heapifyDownForSort(arr, i, 0)
    }
}

fun heapifyDownForSort(arr: IntArray, size: Int, i: Int) {
    var largest = i

    val left = 2 * i + 1
    val right = 2 * i + 2

    if (left < size && arr[left] > arr[largest]) {
        largest = left
    }

    if (right < size && arr[right] > arr[largest]) {
        largest = right
    }

    if (largest != i) {
        val temp = arr[i]
        arr[i] = arr[largest]
        arr[largest] = temp

        heapifyDownForSort(arr, size, largest)
    }
}
