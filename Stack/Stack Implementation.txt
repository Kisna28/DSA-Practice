class StackArrayList{
    val list = ArrayListInt()
    fun isEmpty() Boolean{
        return list.isEmpty()
    }
    push
    fun push(data Int){
        list.add(data)
    }
    pop
    fun pop() Int{
        val top = list.get(list.size-1)
        list.removeLast()
        return top
    }
    peek
    fun peek() Int{
        return list.get(list.size-1)
    }
}
